import {Errors} from '@app/constants';
import * as yup from 'yup';

// 1. Yup Validation Schema
export const PersonalDetailsSchema = yup.object().shape({
  maritalStatus: yup.string().required(Errors.MaritalStatusRequired),
  familyMembers: yup
    .number()
    .required(Errors.FamilyMembersRequired)
    .min(0, Errors.FamilyMembersMin)
    .integer(Errors.FamilyMembersInteger),
  occupation: yup.string().required(Errors.OccupationRequired),
  educationLevel: yup.string().required(Errors.EducationLevelRequired),
  hasAdditionalIncome: yup
    .string()
    .oneOf(['Yes', 'No'], Errors.HasAdditionalIncomeInvalid)
    .required(Errors.HasAdditionalIncomeRequired),

  incomeSource: yup
    .string()
    .when('hasAdditionalIncome', {
      is: 'Yes',
      then: schema => schema.required(Errors.IncomeSourceRequired),
      otherwise: schema => schema.notRequired(),
    }),
  incomeSourceDescription: yup
    .string()
    .when('hasAdditionalIncome', {
      is: 'Yes',
      then: schema =>
        schema
          .min(3, Errors.IncomeSourceDescriptionMinLength)
          .max(100, Errors.IncomeSourceDescriptionMaxLength)
          .required(Errors.IncomeSourceDescriptionRequired),
      otherwise: schema => schema.notRequired(),
    }),
  incomeFrequency: yup
    .string()
    .when('hasAdditionalIncome', {
      is: 'Yes',
      then: schema => schema.required(Errors.IncomeFrequencyRequired),
      otherwise: schema => schema.notRequired(),
    }),
  incomeAmountRange: yup
    .string()
    .when('hasAdditionalIncome', {
      is: 'Yes',
      then: schema =>
        schema
          .matches(
            /^\d{1,3}(,\d{3})*-\d{1,3}(,\d{3})*\sSAR$/,
            Errors.IncomeAmountRangeInvalid,
          ) // Matches "X,XXX-Y,YYY SAR" format
          .required(Errors.IncomeAmountRangeRequired),
      otherwise: schema => schema.notRequired(),
    }),

  isGIBBoardMember: yup
    .string()
    .oneOf(['Yes', 'No'], Errors.IsGIBBoardMemberInvalid)
    .required(Errors.IsGIBBoardMemberRequired),
  boardMemberName: yup
    .string()
    .when('isGIBBoardMember', {
      is: 'Yes',
      then: schema =>
        schema
          .min(3, Errors.BoardMemberNameMinLength)
          .max(50, Errors.BoardMemberNameMaxLength)
          .required(Errors.BoardMemberNameRequired),
      otherwise: schema => schema.notRequired(),
    }),
});
